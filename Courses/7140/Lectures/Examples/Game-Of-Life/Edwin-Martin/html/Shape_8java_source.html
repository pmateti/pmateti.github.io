39  for (int cell = 0; cell < shape.length; cell++) {
40  if (shape[cell][0] > shapeWidth)

42  if (shape[cell][1] > shapeHeight)
43  shapeHeight = shape[cell][1];
44  }
45  shapeWidth++;
46  shapeHeight++;
