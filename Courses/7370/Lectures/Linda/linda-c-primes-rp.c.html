{
  int count = 0, i, is_prime(), ok;
  for(i = 2; i <= LIMIT; ++i) eval("primes", i, is_prime(i));

  for(i = 2; i <= LIMIT; ++i) {
    rd("primes", i, ? ok);
    if (ok) ++count;
  }
  printf("%d.\n", count);
}

is_prime(int me) {
  int
    double
    i, limit, ok;
  sqrt();
  limit = sqrt((double) me) + 1;
  for (i = 2; i < limit; ++i) {
    rd("primes", i, ? ok);
    if (ok && (me%i == 0)) return 0;
  }
  return 1;
}
